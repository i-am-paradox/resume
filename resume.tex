\newcommand{\SkillHexagon}[2]{%
    % #1 = Skill Name, #2 = Level (1-5)
    \begin{tikzpicture}[scale=0.35]
        \foreach \i in {1,...,5} {
            \node[regular polygon,regular polygon sides=6,minimum size=0.5cm,fill=bgLight,draw=borderGray,line width=0.3pt] at (\i*0.55,0) {};
        }
        \foreach \i in {1,...,#2} {
            \node[regular polygon,regular polygon sides=6,minimum size=0.5cm,fill=goldPrimary,draw=goldDark,line width=0.3pt] at (\i*0.55,0) {};
        }
    \end{tikzpicture}%
    \hspace{2pt}\textbf{\footnotesize\color{textDark}#1}%
}

%=================================================================================

\newcommand{\CircularSkill}[3]{%
    % #1 = Label, #2 = Percentage (0-100), #3 = Color
    \begin{tikzpicture}[scale=0.4]
        \draw[line width=2pt,borderGray] (0,0) circle (1);
        \draw[line width=2pt,#3,line cap=round] (90:1) arc (90:90-#2*3.6:1);
        \node[font=\tiny\bfseries,color=textDark] at (0,0) {#2\%};
    \end{tikzpicture}%
    \\ \scriptsize\color{textMid}#1
}
